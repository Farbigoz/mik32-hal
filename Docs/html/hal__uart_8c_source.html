6void HAL_UART_Init(HAL_UART_HandleTypeDef *UartTypeDef) {
7 // Init UART clock
8 if (UartTypeDef->Instance == UART_0)
9 {
10 PM->CLK_APB_P_SET |= PM_CLOCK_UART_0_M;
11 }
12 else if (UartTypeDef->Instance == UART_1)
13 {
14 PM->CLK_APB_P_SET |= PM_CLOCK_UART_1_M;
15 }
16
17 UartTypeDef->State = (HAL_UART_StateTypeDef *)(&UartTypeDef->Instance->FLAGS);
18
19 UartTypeDef->RxBuffPtr = &UartTypeDef->Instance->RXDATA;
20 UartTypeDef->TxBuffPtr = &UartTypeDef->Instance->TXDATA;
21
22 // Disable uart
23 UartTypeDef->Instance->CONTROL1 = 0 << UART_CONTROL1_UE_S;
24
25 // Init.BaudRate
26 UartTypeDef->Instance->DIVIDER = (CPU_CLOCK_MHZ*1000*1000) / UartTypeDef->Init.BaudRate;
27
28 UartTypeDef->Instance->CONTROL1 |= UartTypeDef->Init.TxState;
29 UartTypeDef->Instance->CONTROL1 |= UartTypeDef->Init.RxState;
30
31 UartTypeDef->Instance->CONTROL2 |= UartTypeDef->Init.TxPolarityInvert;
32 UartTypeDef->Instance->CONTROL2 |= UartTypeDef->Init.RxPolarityInvert;
33
34 UartTypeDef->Instance->CONTROL2 |= UartTypeDef->Init.DataPinSwap;
35
36 UartTypeDef->Instance->CONTROL2 |= UartTypeDef->Init.Mode;
37
38 // Enable
39 UartTypeDef->Instance->CONTROL1 |= 1 << UART_CONTROL1_UE_S;
40
41 // Wait to enable
42 while (!(UartTypeDef->State->TxEnable)
43 && !(UartTypeDef->State->RxEnable));
44}
