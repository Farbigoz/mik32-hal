4void HAL_TIMER32_Init(HAL_TIMER32_TypeDef *htim) {
5 uint32_t epicInterruptShift;
6 bool enableEpicInterrupt = false;
7
8 if (htim->Instance == TIMER32_0)
9 {
10 PM->CLK_APB_M_SET |= PM_CLOCK_TIMER32_0_M;
11 epicInterruptShift = EPIC_TIMER32_0_INDEX;
12 }
13 else if (htim->Instance == TIMER32_1)
14 {
15 PM->CLK_APB_P_SET |= PM_CLOCK_TIMER32_1_M;
16 epicInterruptShift = EPIC_TIMER32_1_INDEX;
17 }
18 else if (htim->Instance == TIMER32_2)
19 {
20 PM->CLK_APB_P_SET |= PM_CLOCK_TIMER32_2_M;
21 epicInterruptShift = EPIC_TIMER32_2_INDEX;
22 }
23 else
24 {
25 return;
26 }
27
28 htim->Instance->Enable = TIMER32_RESET_VALUE_M;
29
30 htim->Instance->Prescaler = htim->Init.Prescaler;
31 htim->Instance->Top = htim->Init.Period;
32
33 htim->Instance->Control = 0x0;
34 htim->Instance->Control |= htim->Init.CountMode << TIMER32_CONTROL_MODE_S;
35
36 htim->Instance->IntMask = 0x0;
37 htim->Instance->IntClear = 0xffffffff;
38
39 htim->Instance->IntMask = htim->Init.Interrupt;
40 enableEpicInterrupt = htim->Init.Interrupt != HAL_TIMER32_INTERRUPT_NONE;
41
42 for (uint8_t i = 0; i < TIMER32_CHANNEL_COUNT; i++) {
43 htim->Instance->Channels[i].Control = 0x0;
44 htim->Instance->Channels[i].Control |= htim->Channel[i].Mode << TIMER32_CH_CONTROL_MODE_S;
45
46 htim->Instance->IntMask |= htim->Channel[i].Interrupt << i;
47 enableEpicInterrupt |= htim->Channel[i].Interrupt != HAL_TIMER32_CHANNEL_INTERRUPT_NONE;
48
49 switch (htim->Channel[i].Mode) {
50 case HAL_TIMER32_CHANNEL_MODE_CAPTURE:
51 htim->Instance->Channels[i].Control |= htim->Channel[i].CaptureMode << TIMER32_CH_CONTROL_CAPTURE_EDGE_S;
52 break;
53
54 case HAL_TIMER32_CHANNEL_MODE_PWM:
55 htim->Instance->Channels[i].Control |= htim->Channel[i].PwmType << TIMER32_CH_CONTROL_INVERTED_PWM_S;
56
57 case HAL_TIMER32_CHANNEL_MODE_COMPARE:
58 htim->Instance->Channels[i].OCR = htim->Channel[i].Compare;
59 break;
60 }
61 }
62
63 if (enableEpicInterrupt) {
64 EPIC->MASK_SET = 1 << epicInterruptShift;
65 }
66}
